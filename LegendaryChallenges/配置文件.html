配置文件 ​

config.yml 核心配置文件
js
store:
  #Yaml / Mysql 默认 Yaml
  mode: Yaml
  #玩家数据、公会排行数据更新自动保存间隔。秒为单位
  updata: 600
Mysql:
  address: localhost
  port: 3306
  database: test
  user: root
  password: 123456
HikariCP:
  connectionTimeout: 30000
  #连接超时时间 默认值：30000（30秒）
  minimumIdle: 10
  #最小空闲连接数
  maximumPoolSize: 50
  #池允许达到的最大大小，包括空闲和正在使用的连接
set:
  sound:
    #任务完成时候的音效
    complete_everyday: UI_TOAST_CHALLENGE_COMPLETE
  completed_all:
    run:
      - '[console]money give %player% 50'
